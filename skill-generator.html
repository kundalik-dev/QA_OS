Open claude/skills/skills-config.js and append a new object to the SKILLS_CONFIG array. That's the only file you need to touch — this HTML never needs editing. 

{
  id:          'my-skill',          // unique identifier
  name:        'My Skill',          // tab label
  icon:        '🚀',                // emoji shown in tab + header
  color:       '#a78bfa',           // accent hex color
  description: 'What this skill does',
  command:     '/my.command',       // the trigger prefix for Claude
  fields: [
    // Free-text input
    { id: 'title', label: 'Title', type: 'text',
      required: true, placeholder: 'e.g. Something here',
      span: 'full' },               // span:'full' = full row width

    // Dropdown (known/fixed values)
    { id: 'status', label: 'Status', type: 'select',
      options: ['Open', 'Closed'], default: 'Open' },

    // Text input with hint
    { id: 'due', label: 'Due Date', type: 'text',
      placeholder: 'e.g. 15march, friday',
      hint: 'Natural language date is fine' },
  ],
  generate(vals) {
    // Return the final prompt string
    let cmd = `/my.command "${vals.title}"`;
    if (vals.status) cmd += ` status=${vals.status}`;
    if (vals.due)    cmd += ` due=${vals.due}`;
    return cmd;
  }
}

Save skills-config.js and refresh the browser — the new tab appears automatically.

Paste the generated prompt directly into Claude Code to invoke the skill.
